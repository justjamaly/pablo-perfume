<script>
let category='all';
let cart=[];

const products=[
  {name:"Dior Sauvage",price:85000,category:"men",image:"images/sauvage.jpg",description:"Fresh and spicy with notes of bergamot and pepper."},
  {name:"Bleu de Chanel",price:90000,category:"men",image:"images/bleu.jpg",description:"Woody aromatic fragrance, elegant and refined.",discount:80000,new:true},
  {name:"Oud Wood",price:120000,category:"unisex",image:"images/oudwood.jpg",description:"Exotic oud with rich woody notes."},
  {name:"Chanel Coco Mademoiselle",price:95000,category:"women",image:"images/coco.jpg",description:"Classic floral scent with orange and jasmine.",discount:85000,new:true},
  {name:"Gucci Bloom",price:80000,category:"women",image:"images/bloom.jpg",description:"Soft floral with notes of tuberose and jasmine."},
  {name:"Tom Ford Noir",price:110000,category:"men",image:"images/tomford.jpg",description:"Dark, sensual fragrance with warm spicy notes."}
];

function renderProducts(){
  const grid=document.getElementById('products');
  grid.innerHTML='';
  products.forEach((p,index)=>{
    let card=document.createElement('div');
    card.className='card '+p.category;
    card.dataset.name=p.name;
    card.dataset.price=p.price;
    card.innerHTML=`${p.new?'<div class="badge">NEW</div>':''}
      <img src="${p.image}" loading="lazy">
      <h3>${p.name}</h3>
      <p class="price">TZS ${p.price}</p>
      <button onclick="addToCart(${index})">Add to Cart</button>`;
    grid.appendChild(card);
  });
  filterProducts();
}

function setCategory(c){category=c;filterProducts();}
document.getElementById('search').onkeyup=filterProducts;
document.getElementById('sort').onchange=sortProducts;

function filterProducts(){
  let q=document.getElementById('search').value.toLowerCase();
  document.querySelectorAll('.card').forEach(c=>{
    let match=c.dataset.name.toLowerCase().includes(q);
    let cat=category==='all'||c.classList.contains(category);
    c.style.display=(match&&cat)?'block':'none';
  });
}

function sortProducts(){
  let grid=document.getElementById('products');
  let cards=[...grid.children];
  let mode=document.getElementById('sort').value;
  if(mode!=='default'){
    cards.sort((a,b)=>{
      return mode==='low'?a.dataset.price-b.dataset.price:b.dataset.price-a.dataset.price;
    });
    cards.forEach(c=>grid.appendChild(c));
  }
}

function addToCart(index){
  let item={...products[index],qty:1};
  let exist=cart.find(i=>i.name===item.name);
  if(exist){exist.qty++;}else{cart.push(item);}
  updateCart();
  openCart();
}

function updateCart(){
  const cartItems=document.getElementById('cartItems');
  const cartTotal=document.getElementById('cartTotal');
  cartItems.innerHTML='';
  let total=0;
  cart.forEach((i,index)=>{
    total+=i.price*i.qty;
    let div=document.createElement('div');
    div.className='cart-item';
    div.innerHTML=`<strong>${i.name}</strong>
      <span>Price: TZS ${i.price}</span>
      <span>Qty: ${i.qty}</span>
      <div class="qty">
        <button data-index="${index}" class="inc">+</button>
        <button data-index="${index}" class="dec">-</button>
      </div>`;
    cartItems.appendChild(div);
  });
  cartTotal.textContent=total;
  document.querySelectorAll('.inc').forEach(b=>b.addEventListener('click',()=>{ cart[b.dataset.index].qty++; updateCart(); }));
  document.querySelectorAll('.dec').forEach(b=>b.addEventListener('click',()=>{ cart[b.dataset.index].qty--; if(cart[b.dataset.index].qty<=0) cart.splice(b.dataset.index,1); updateCart(); }));
}

function openCart(){document.getElementById('cartDrawer').style.right='0';}
function closeCart(){document.getElementById('cartDrawer').style.right='-400px';}

function checkout(){
  if(!cart.length){alert('Cart is empty');return;}
  let msg="Hello Pablo Perfumes,%0A%0AI want to order:%0A";
  let total=0;
  cart.forEach(i=>{msg+=`- ${i.name} x ${i.qty} (TZS ${i.price})%0A`; total+=i.price*i.qty;});
  msg+=`%0ATotal: TZS ${total}`;
  window.open("https://wa.me/255675999999?text="+msg,"_blank");
}

renderProducts();
document.getElementById('cartBtn').addEventListener('click',openCart);
