<script>
const searchInput = document.getElementById('search');
const sortSelect = document.getElementById('sort');
const categoryButtons = document.querySelectorAll('.controls button[data-category]');
const productsGrid = document.getElementById('products');
const cartBtn = document.getElementById('cartBtn');

let category = 'all';
let cart = [];

// Filter products
const filterProducts = () => {
  const query = searchInput.value.toLowerCase();
  [...productsGrid.children].forEach(card => {
    const matchesSearch = card.dataset.name.toLowerCase().includes(query);
    const matchesCategory = category === 'all' || card.classList.contains(category);
    card.style.display = (matchesSearch && matchesCategory) ? 'block' : 'none';
  });
};

// Sort products
const sortProducts = () => {
  const mode = sortSelect.value;
  const cards = [...productsGrid.children];
  if(mode !== 'default') {
    cards.sort((a, b) => mode === 'low' 
      ? a.dataset.price - b.dataset.price 
      : b.dataset.price - a.dataset.price);
    cards.forEach(c => productsGrid.appendChild(c));
  }
};

// Set category
categoryButtons.forEach(btn => btn.addEventListener('click', () => {
  category = btn.dataset.category;
  filterProducts();
}));

// Add to cart using event delegation
productsGrid.addEventListener('click', e => {
  if(e.target.classList.contains('add-btn')) {
    const card = e.target.closest('.card');
    const name = card.dataset.name;
    const price = parseInt(card.dataset.price);
    cart.push({name, price});
    alert(`${name} added to cart`);
  }
});

// Checkout
cartBtn.addEventListener('click', () => {
  if(!cart.length) { alert("Cart is empty"); return; }
  let msg = "Hello Pablo Perfumes,%0A%0AI want to order:%0A";
  let total = 0;
  cart.forEach(item => { 
    msg += `- ${item.name} (TZS ${item.price})%0A`; 
    total += item.price;
  });
  msg += `%0ATotal: TZS ${total}`;
  window.open(`https://wa.me/255675999999?text=${msg}`, "_blank");
});

searchInput.addEventListener('input', filterProducts);
sortSelect.addEventListener('change', sortProducts);
</script>
