<script>
const products = [
  { name: "Dior Sauvage", price: 85000, category: "men", image: "images/sauvage.jpg", description:"Fresh and spicy with notes of bergamot and pepper." },
  { name: "Bleu de Chanel", price: 90000, category: "men", image: "images/bleu.jpg", description:"Woody aromatic fragrance, elegant and refined.", discount: 80000 },
  { name: "Oud Wood", price: 120000, category: "unisex", image: "images/oudwood.jpg", description:"Exotic oud with rich woody notes." },
  { name: "Chanel Coco Mademoiselle", price: 95000, category: "women", image: "images/coco.jpg", description:"Classic floral scent with orange and jasmine.", discount: 85000 },
  { name: "Gucci Bloom", price: 80000, category: "women", image: "images/bloom.jpg", description:"Soft floral with notes of tuberose and jasmine." },
  { name: "Tom Ford Noir", price: 110000, category: "men", image: "images/tomford.jpg", description:"Dark, sensual fragrance with warm spicy notes." }
];

let category = 'all', cart = [], selectedProduct = null;
const productsGrid = document.getElementById('products');
const searchInput = document.getElementById('search');
const sortSelect = document.getElementById('sort');
const categoryButtons = document.querySelectorAll('.controls button[data-category]');
const modal = document.getElementById('modal');
const modalImg = modal.querySelector('img');
const modalTitle = modal.querySelector('h3');
const modalDesc = modal.querySelector('p');
const modalAdd = document.getElementById('modalAdd');
const modalClose = document.getElementById('modalClose');
const cartBtn = document.getElementById('cartBtn');
const cartCount = document.getElementById('cartCount');
const cartDrawer = document.getElementById('cartDrawer');
const drawerClose = document.getElementById('drawerClose');
const cartItems = document.getElementById('cartItems');
const cartTotal = document.getElementById('cartTotal');
const checkoutBtn = document.getElementById('checkoutBtn');
const noProducts = document.getElementById('noProducts');

function renderProducts(){
  productsGrid.innerHTML='';
  let filtered = products.filter(p => (category==='all'||p.category===category)&&p.name.toLowerCase().includes(searchInput.value.toLowerCase()));
  if(sortSelect.value==='low') filtered.sort((a,b)=>(a.discount||a.price)-(b.discount||b.price));
  if(sortSelect.value==='high') filtered.sort((a,b)=>(b.discount||b.price)-(a.discount||a.price));
  if(!filtered.length){ noProducts.style.display='block'; return; } else { noProducts.style.display='none'; }
  filtered.forEach(p=>{
    const card=document.createElement('div'); card.className=`card ${p.category}`; card.dataset.name=p.name; card.dataset.price=p.discount||p.price;
    card.innerHTML=`<div class="badge">${p.category}</div>
      ${p.discount?'<div class="badge discount-badge">-'+Math.round((1-p.discount/p.price)*100)+'%</div>':''}
      <img src="${p.image}" alt="${p.name}" loading="lazy" onerror="this.src='images/fallback.jpg'">
      <h3>${p.name}</h3>
      <p class="price">TZS ${(p.discount||p.price).toLocaleString()} ${p.discount?'<span class="discount">TZS '+p.price.toLocaleString()+'</span>':''}</p>`;
    productsGrid.appendChild(card);
    card.addEventListener('click',()=>{ selectedProduct=p; modalImg.src=p.image; modalTitle.textContent=p.name; modalDesc.textContent=p.description; modal.style.display='flex'; });
  });
}

searchInput.addEventListener('input',renderProducts);
sortSelect.addEventListener('change',renderProducts);
categoryButtons.forEach(btn=>btn.addEventListener('click',()=>{category=btn.dataset.category; renderProducts();}));

modalClose.addEventListener('click',()=>{modal.style.display='none';});
modalAdd.addEventListener('click',()=>{
  if(selectedProduct){ cart.push({name:selectedProduct.name,price:selectedProduct.discount||selectedProduct.price}); updateCart(); modal.style.display='none'; }
});
window.addEventListener('click', e=>{if(e.target===modal) modal.style.display='none';});

cartBtn.addEventListener('click',()=>{cartDrawer.style.right='0';});
drawerClose.addEventListener('click',()=>{cartDrawer.style.right='-100%';});

function updateCart(){
  cartCount.textContent=cart.length;
  cartItems.innerHTML='';
  let total=0;
  cart.forEach((item,index)=>{
    total+=item.price;
    const div=document.createElement('div'); div.className='item';
    div.innerHTML=`<span>${item.name} (TZS ${item.price.toLocaleString()})</span><button data-index="${index}">X</button>`;
    cartItems.appendChild(div);
  });
  cartTotal.textContent=total.toLocaleString();
  cartItems.querySelectorAll('button').forEach(btn=>btn.addEventListener('click',()=>{ cart.splice(btn.dataset.index,1); updateCart(); }));
}

checkoutBtn.addEventListener('click',()=>{
  if(!cart.length){ alert("Cart is empty"); return; }
  let msg="Hello Pablo Perfumes,%0A%0AI want to order:%0A"; let total=0;
  cart.forEach(item=>{ msg+=`- ${item.name} (TZS ${item.price})%0A`; total+=item.price; });
  msg+=`%0ATotal: TZS ${total}`; window.open(`https://wa.me/255675999999?text=${msg}`,"_blank");
});

renderProducts();
